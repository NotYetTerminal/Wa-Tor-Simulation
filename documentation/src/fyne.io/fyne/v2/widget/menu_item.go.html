}

func ( *menuItem) () bool {
	if .Child() == nil {
		return false
	}
	if .isSubmenuOpen() {
		return .Child().ActivateLastSubmenu()
	}
	.Child().Show()
	.Child().ActivateNext()
	return true
}

func ( *menuItem) () {
	if .Child() != nil {
		.Child().Hide()
	}
	.Parent.DeactivateChild()
}

func ( *menuItem) () bool {
	if !.isSubmenuOpen() {
		return false
	}
	if !.Child().DeactivateLastSubmenu() {
		.Child().DeactivateChild()
		.Child().Hide()
	}
	return true
}
