// CreateRenderer is a private method to Fyne which links this widget to its renderer
func ( *Label) () fyne.WidgetRenderer {
	.provider = NewRichTextWithText(.Text)
	.ExtendBaseWidget()
	.syncSegments()

	return NewSimpleRenderer(.provider)
}

// MinSize returns the size that this label should not shrink below.
//
// Implements: fyne.Widget
func ( *Label) () fyne.Size {
	.ExtendBaseWidget()
	return .BaseWidget.MinSize()
}

// Refresh triggers a redraw of the label.
//
// Implements: fyne.Widget
func ( *Label) () {
	if .provider == nil { // not created until visible
		return
	}
	.syncSegments()
	.provider.Refresh()
	.BaseWidget.Refresh()
}

// Resize sets a new size for the label.
// This should only be called if it is not in a container with a layout manager.
//
// Implements: fyne.Widget
func ( *Label) ( fyne.Size) {
	.BaseWidget.Resize()
	if .provider != nil {
		.provider.Resize()
	}
}

func ( *Label) () {
	.propertyLock.RLock()
	defer .propertyLock.RUnlock()

	var  fyne.ThemeColorName
	switch .Importance {
	case LowImportance:
		 = theme.ColorNameDisabled
	case MediumImportance:
		 = theme.ColorNameForeground
	case HighImportance:
		 = theme.ColorNamePrimary
	case DangerImportance:
		 = theme.ColorNameError
	case WarningImportance:
		 = theme.ColorNameWarning
	case SuccessImportance:
		 = theme.ColorNameSuccess
	default:
		 = theme.ColorNameForeground
	}

	.provider.Wrapping = .Wrapping
	.provider.Truncation = .Truncation
	.provider.Segments[0].(*TextSegment).Style = RichTextStyle{
		Alignment: .Alignment,
		ColorName: ,
		Inline:    true,
		TextStyle: .TextStyle,
	}
	.provider.Segments[0].(*TextSegment).Text = .Text
}
