// TypedKey is called if a key event happens while this List is focused.

// Implements: fyne.Focusable

	case fyne.KeySpace:
		.Select(.currentFocus)
	case fyne.KeyDown:
		if  := .Length;  != nil && .currentFocus >= ()-1 {
			return
		}

		.currentFocus++
		.scrollTo(.currentFocus)
		.RefreshItem(.currentFocus)
	case fyne.KeyUp:
		if .currentFocus <= 0 {
			return
		}
		.RefreshItem(.currentFocus)
		.currentFocus--
		.scrollTo(.currentFocus)
		.RefreshItem(.currentFocus)
	}
}
