// HyperlinkSegment represents a hyperlink within a rich text widget.
//
// Since: 2.1
type HyperlinkSegment struct {
	Alignment fyne.TextAlign
	Text      string
	URL       *url.URL

	// OnTapped overrides the default `fyne.OpenURL` call when the link is tapped
	//
	// Since: 2.4
	OnTapped func() `json:"-"`
}

// Inline returns true as hyperlinks are inside other elements.
func ( *HyperlinkSegment) () bool {
	return true
}

// Textual returns the content of this segment rendered to plain text.
func ( *HyperlinkSegment) () string {
	return .Text
}

// Visual returns the hyperlink widget required to render this segment.
func ( *HyperlinkSegment) () fyne.CanvasObject {
	 := NewHyperlink(.Text, .URL)
	.Alignment = .Alignment
	.OnTapped = .OnTapped
	return &fyne.Container{Layout: &unpadTextWidgetLayout{parent: }, Objects: []fyne.CanvasObject{}}
}

// Update applies the current state of this hyperlink segment to an existing visual.
func ( *HyperlinkSegment) ( fyne.CanvasObject) {
	 := .(*fyne.Container).Objects[0].(*Hyperlink)
	.Text = .Text
	.URL = .URL
	.Alignment = .Alignment
	.OnTapped = .OnTapped
	.Refresh()
}
