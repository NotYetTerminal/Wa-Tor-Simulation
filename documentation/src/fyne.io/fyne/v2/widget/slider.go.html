// TypedKey is called when this item receives a key event.

func ( *Slider) ( *fyne.KeyEvent) {
	if .disabled {
		return
	}
	if .Orientation == Vertical {
		switch .Name {
		case fyne.KeyUp:
			.SetValue(.Value + .Step)
		case fyne.KeyDown:
			.SetValue(.Value - .Step)
		}
	} else {
		switch .Name {
		case fyne.KeyLeft:
			.SetValue(.Value - .Step)
		case fyne.KeyRight:
			.SetValue(.Value + .Step)
		}
	}
}
