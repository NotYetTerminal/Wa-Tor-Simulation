// Tapped is called when a pointer tapped event is captured and triggers any change handler
func ( *Check) ( *fyne.PointEvent) {
	if .Disabled() {
		return
	}
	if !.minSize.IsZero() &&
		(.Position.X > .minSize.Width || .Position.Y > .minSize.Height) {
		// tapped outside the active area of the widget
		return
	}

	if !.focused {
		focusIfNotMobile(.super())
	}
	.SetChecked(!.Checked)
}
