func ( *Form) () {
	if .CancelText == "" {
		.CancelText = "Cancel"
	}
	if .SubmitText == "" {
		.SubmitText = "Submit"
	}

	// set visibility on the buttons
	if .OnCancel == nil {
		.cancelButton.Hide()
	} else {
		.cancelButton.SetText(.CancelText)
		.cancelButton.OnTapped = .OnCancel
		.cancelButton.Show()
	}
	if .OnSubmit == nil {
		.submitButton.Hide()
	} else {
		.submitButton.SetText(.SubmitText)
		.submitButton.OnTapped = .OnSubmit
		.submitButton.Show()
	}
	if .OnCancel == nil && .OnSubmit == nil {
		.buttonBox.Hide()
	} else {
		.buttonBox.Show()
	}
}
