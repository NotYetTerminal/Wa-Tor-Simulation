// ProgressBarInfinite widget creates a horizontal panel that indicates waiting indefinitely
// An infinite progress bar loops 0% -> 100% repeatedly until Stop() is called
type ProgressBarInfinite struct {
	BaseWidget
	running bool
}

// Show this widget, if it was previously hidden
func ( *ProgressBarInfinite) () {
	.propertyLock.Lock()
	.running = true
	.propertyLock.Unlock()

	.BaseWidget.Show()
}

// Hide this widget, if it was previously visible
func ( *ProgressBarInfinite) () {
	.propertyLock.Lock()
	.running = false
	.propertyLock.Unlock()

	.BaseWidget.Hide()
}

// Start the infinite progress bar animation
func ( *ProgressBarInfinite) () {
	.propertyLock.Lock()
	if .running {
		.propertyLock.Unlock()
		return
	}

	.running = true
	.propertyLock.Unlock()

	.BaseWidget.Refresh()
}

// Stop the infinite progress bar animation
func ( *ProgressBarInfinite) () {
	.propertyLock.Lock()
	if !.running {
		.propertyLock.Unlock()
		return
	}

	.running = false
	.propertyLock.Unlock()

	.BaseWidget.Refresh()
}

// Running returns the current state of the infinite progress animation
func ( *ProgressBarInfinite) () bool {
	.propertyLock.RLock()
	defer .propertyLock.RUnlock()
	return .running
}

// MinSize returns the size that this widget should not shrink below
func ( *ProgressBarInfinite) () fyne.Size {
	.ExtendBaseWidget()
	return .BaseWidget.MinSize()
}
