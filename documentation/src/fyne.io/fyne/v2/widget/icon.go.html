type iconRenderer struct {
	widget.BaseRenderer
	raster *canvas.Image

	image *Icon
}

func ( *iconRenderer) () fyne.Size {
	return fyne.NewSquareSize(.image.Theme().Size(theme.SizeNameInlineIcon))
}

func ( *iconRenderer) ( fyne.Size) {
	if len(.Objects()) == 0 {
		return
	}

	.Objects()[0].Resize()
}

func ( *iconRenderer) () {
	if .image.Resource == .image.cachedRes {
		return
	}

	.image.propertyLock.RLock()
	.raster.Resource = .image.Resource
	.image.cachedRes = .image.Resource

	if .image.Resource == nil {
		.raster.Image = nil // reset the internal caching too...
	}
	.image.propertyLock.RUnlock()

	.raster.Refresh()
}
