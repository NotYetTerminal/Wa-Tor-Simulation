// PopUpMenu is a Menu which displays itself in an OverlayContainer.
type PopUpMenu struct {
	*Menu
	canvas  fyne.Canvas
	overlay *widget.OverlayContainer
}

// NewPopUpMenu creates a new, reusable popup menu. You can show it using ShowAtPosition.
//
// Since: 2.0
func ( *fyne.Menu,  fyne.Canvas) *PopUpMenu {
	 := &Menu{}
	.setMenu()
	 := &PopUpMenu{Menu: , canvas: }
	.ExtendBaseWidget()
	.Menu.Resize(.Menu.MinSize())
	.Menu.customSized = true
	 := widget.NewOverlayContainer(, , .Dismiss)
	.Resize(.MinSize())
	.overlay = 
	.OnDismiss = func() {
		.Hide()
	}
	return 
}

// Show makes the pop-up menu visible.
//
// Implements: fyne.Widget
func ( *PopUpMenu) () {
	.Menu.alignment = .alignment
	.Menu.Refresh()

	.overlay.Show()
	.Menu.Show()
	if !fyne.CurrentDevice().IsMobile() {
		.canvas.Focus()
	}
}

// ShowAtPosition shows the pop-up menu at the specified position.
func ( *PopUpMenu) ( fyne.Position) {
	.Move()
	.Show()
}

}

// TypedKey handles key events. It allows keyboard control of the pop-up menu.
//
// Implements: fyne.Focusable
func ( *PopUpMenu) ( *fyne.KeyEvent) {
	switch .Name {
	case fyne.KeyDown:
		.ActivateNext()
	case fyne.KeyEnter, fyne.KeyReturn, fyne.KeySpace:
		.TriggerLast()
	case fyne.KeyEscape:
		.Dismiss()
	case fyne.KeyLeft:
		.DeactivateLastSubmenu()
	case fyne.KeyRight:
		.ActivateLastSubmenu()
	case fyne.KeyUp:
		.ActivatePrevious()
	}
}
