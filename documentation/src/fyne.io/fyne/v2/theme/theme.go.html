type builtinTheme struct {
	variant fyne.ThemeVariant

	regular, bold, italic, boldItalic, monospace, symbol fyne.Resource
}

func ( *builtinTheme) () {
	.regular = regular
	.bold = bold
	.italic = italic
	.boldItalic = bolditalic
	.monospace = monospace
	.symbol = symbol

	 := os.Getenv("FYNE_FONT")
	if  != "" {
		.regular = loadCustomFont(, "Regular", regular)
		if .regular == regular { // failed to load
			.bold = loadCustomFont(, "Bold", bold)
			.italic = loadCustomFont(, "Italic", italic)
			.boldItalic = loadCustomFont(, "BoldItalic", bolditalic)
		} else { // first custom font loaded, fall back to that
			.bold = loadCustomFont(, "Bold", .regular)
			.italic = loadCustomFont(, "Italic", .regular)
			.boldItalic = loadCustomFont(, "BoldItalic", .regular)
		}
	}
	 = os.Getenv("FYNE_FONT_MONOSPACE")
	if  != "" {
		.monospace = loadCustomFont(, "Regular", monospace)
	}
	 = os.Getenv("FYNE_FONT_SYMBOL")
	if  != "" {
		.symbol = loadCustomFont(, "Regular", symbol)
	}
}

func ( *builtinTheme) ( fyne.TextStyle) fyne.Resource {
	if .Monospace {
		return .monospace
	}
	if .Bold {
		if .Italic {
			return .boldItalic
		}
		return .bold
	}
	if .Italic {
		return .italic
	}
	if .Symbol {
		return .symbol
	}
	return .regular
}
