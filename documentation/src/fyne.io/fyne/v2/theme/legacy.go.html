// FromLegacy returns a 2.0 Theme created from the given LegacyTheme data.
// This is a transition path and will be removed in the future (probably version 3.0).
//
// Since: 2.0
func ( fyne.LegacyTheme) fyne.Theme {
	return &legacyWrapper{old: }
}

var _ fyne.Theme = (*legacyWrapper)(nil)

type legacyWrapper struct {
	old fyne.LegacyTheme
}

func ( *legacyWrapper) ( fyne.ThemeColorName,  fyne.ThemeVariant) color.Color {
	switch  {
	case ColorNameBackground:
		return .old.BackgroundColor()
	case ColorNameForeground:
		return .old.TextColor()
	case ColorNameButton:
		return .old.ButtonColor()
	case ColorNameDisabledButton:
		return .old.DisabledButtonColor()
	case ColorNameDisabled:
		return .old.DisabledTextColor()
	case ColorNameFocus:
		return .old.FocusColor()
	case ColorNameHover:
		return .old.HoverColor()
	case ColorNamePlaceHolder:
		return .old.PlaceHolderColor()
	case ColorNamePrimary:
		return .old.PrimaryColor()
	case ColorNameScrollBar:
		return .old.ScrollBarColor()
	case ColorNameShadow:
		return .old.ShadowColor()
	default:
		return DefaultTheme().Color(, )
	}
}

func ( *legacyWrapper) ( fyne.TextStyle) fyne.Resource {
	if .Monospace {
		return .old.TextMonospaceFont()
	}
	if .Bold {
		if .Italic {
			return .old.TextBoldItalicFont()
		}
		return .old.TextBoldFont()
	}
	if .Italic {
		return .old.TextItalicFont()
	}
	return .old.TextFont()
}

func ( *legacyWrapper) ( fyne.ThemeIconName) fyne.Resource {
	return DefaultTheme().Icon()
}

func ( *legacyWrapper) ( fyne.ThemeSizeName) float32 {
	switch  {
	case SizeNameInlineIcon:
		return float32(.old.IconInlineSize())
	case SizeNamePadding:
		return float32(.old.Padding())
	case SizeNameScrollBar:
		return float32(.old.ScrollBarSize())
	case SizeNameScrollBarSmall:
		return float32(.old.ScrollBarSmallSize())
	case SizeNameText:
		return float32(.old.TextSize())
	default:
		return DefaultTheme().Size()
	}
}
