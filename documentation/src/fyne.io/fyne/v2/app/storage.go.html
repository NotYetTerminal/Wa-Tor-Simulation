type store struct {
	*internal.Docs
	a *fyneApp
}

func ( *store) () fyne.URI {
	if .a.UniqueID() == "" {
		fyne.LogError("Storage API requires a unique ID, use app.NewWithID()", nil)
		return storage.NewFileURI(os.TempDir())
	}

	return storage.NewFileURI(.a.storageRoot())
}

func ( *store) () (fyne.URI, error) {
	return storage.Child(.RootURI(), "Documents")
}
