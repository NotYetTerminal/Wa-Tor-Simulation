// fetch color variant from dbus portal desktop settings.
func () fyne.ThemeVariant {
	,  := appearance.GetColorScheme()
	if  != nil {
		return theme.VariantDark
	}

	return colorSchemeToThemeVariant()
}

func ( appearance.ColorScheme) fyne.ThemeVariant {
	switch  {
	case appearance.Light:
		return theme.VariantLight
	case appearance.Dark:
		return theme.VariantDark
	}

	// Default to light theme to support Gnome's default see https://github.com/fyne-io/fyne/pull/3561
	return theme.VariantLight
}

func ( *fyneApp) ( *fyne.Notification) {
	if build.IsFlatpak {
		 := .sendNotificationThroughPortal()
		if  != nil {
			fyne.LogError("Sending notification using portal failed", )
		}
		return
	}

	,  := dbus.SessionBus() // shared connection, don't close
	if  != nil {
		fyne.LogError("Unable to connect to session D-Bus", )
		return
	}

	 := .cachedIconPath()
	 := int32(0) // we don't support this yet

	 := .Object("org.freedesktop.Notifications", "/org/freedesktop/Notifications")
	 := .Call("org.freedesktop.Notifications.Notify", 0, .uniqueID, uint32(0),
		, .Title, .Content, []string{}, map[string]dbus.Variant{}, )
	if .Err != nil {
		fyne.LogError("Failed to send message to bus", .Err)
	}
}

// Sending with same ID replaces the old notification.
var notificationID atomic.Uint64

// See https://flatpak.github.io/xdg-desktop-portal/docs/#gdbus-org.freedesktop.portal.Notification.
func ( *fyneApp) ( *fyne.Notification) error {
	return notification.Add(
		uint(notificationID.Add(1)),
		notification.Content{
			Title: .Title,
			Body:  .Content,
			Icon:  .uniqueID,
		},
	)
}

func ( *settings) {
	go func() {
		// Theme lookup hangs on some desktops. Update theme variant cache from within goroutine.
		internalapp.CurrentVariant.Store(uint64(findFreedesktopColorScheme()))
		.setupTheme()

		portalSettings.OnSignalSettingChanged(func( portalSettings.Changed) {
			if .Namespace == "org.freedesktop.appearance" && .Key == "color-scheme" {
				 := colorSchemeToThemeVariant(appearance.ColorScheme(.Value.(uint32)))
				internalapp.CurrentVariant.Store(uint64())
				.setupTheme()
			}
		})
	}()
}
