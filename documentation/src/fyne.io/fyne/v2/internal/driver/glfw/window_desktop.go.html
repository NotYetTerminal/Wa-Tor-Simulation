func ( string) fyne.KeyName {

	}

	case '+':

func ( *window) () {
	runOnMain(func() {

		} else {
			glfw.WindowHint(glfw.Decorated, glfw.False)
		}
		if .fixedSize {
			glfw.WindowHint(glfw.Resizable, glfw.False)
		} else {
			glfw.WindowHint(glfw.Resizable, glfw.True)
		}
		glfw.WindowHint(glfw.AutoIconify, glfw.False)
		initWindowHints()

		,  := .screenSize(.canvas.size)
		 = int(fyne.Max(float32(), float32(.width)))
		if  == 0 {
			 = 10
		}
		 = int(fyne.Max(float32(), float32(.height)))
		if  == 0 {
			 = 10
		}

		,  := glfw.CreateWindow(, , .title, nil, nil)
		if  != nil {
			.driver.initFailed("window creation error", )
			return
		}

		.viewLock.Lock()
		.viewport = 
		.viewLock.Unlock()
	})
	if .view() == nil { // something went wrong above, it will have been logged
		return
	}

	// run the GL init on the draw thread
	runOnDraw(, func() {
		.canvas.SetPainter(gl.NewPainter(.canvas, ))
		.canvas.Painter().Init()
	})

	runOnMain(func() {
		.setDarkMode()

		 := .view()
		.SetCloseCallback(.closed)
		.SetPosCallback(.moved)
		.SetSizeCallback(.resized)

		.SetCursorPosCallback(.mouseMoved)
		.SetMouseButtonCallback(.mouseClicked)
		.SetScrollCallback(.mouseScrolled)

		.SetCharCallback(.charInput)
		.SetFocusCallback(.focused)

		.canvas.detectedScale = .detectScale()
		.canvas.scale = .calculatedScale()
		.canvas.texScale = .detectTextureScale()

		.fitContent()

			 := .canvas.canvasSize(.canvas.Content().MinSize())

			.shouldWidth, .shouldHeight = .width, .height
		}

		.requestedWidth, .requestedHeight = .width, .height
		// order of operation matters so we do these last items in order
		.viewport.SetSize(.shouldWidth, .shouldHeight) // ensure we requested latest size
	})
}
