// MouseIn activates the item and shows the menu if the bar is active.
// The menu that was displayed before will be hidden.
//
// If the bar is not active, the item will be hovered.
//
// Implements: desktop.Hoverable
func ( *menuBarItem) ( *desktop.MouseEvent) {
	.hovered = true
	if .Parent.active {
		.Parent.canvas.Focus()
	}
	.Refresh()
}

// MouseMoved activates the item and shows the menu if the bar is active.
// The menu that was displayed before will be hidden.
// This might have an effect when mouse and keyboard control are mixed.
// Changing the active menu with the keyboard will make the hovered menu bar item inactive.
// On the next mouse move the hovered item is activated again.
//
// If the bar is not active, this will do nothing.
//
// Implements: desktop.Hoverable
func ( *menuBarItem) ( *desktop.MouseEvent) {
	if .Parent.active {
		.Parent.canvas.Focus()
	}
}

// MouseOut does nothing if the bar is active.
//
// IF the bar is not active, it changes the item to not be hovered.
//
// Implements: desktop.Hoverable
func ( *menuBarItem) () {
	.hovered = false
	.Refresh()
}

// Tapped toggles the activation state of the menu bar.
// It shows the item’s menu if the bar is activated and hides it if the bar is deactivated.
//
// Implements: fyne.Tappable
func ( *menuBarItem) (*fyne.PointEvent) {
	.Parent.toggle()
}

func ( *menuBarItem) ( *fyne.KeyEvent) {
	switch .Name {
	case fyne.KeyLeft:
		if !.Child().DeactivateLastSubmenu() {
			.Parent.canvas.FocusPrevious()
		}
	case fyne.KeyRight:
		if !.Child().ActivateLastSubmenu() {
			.Parent.canvas.FocusNext()
		}
	case fyne.KeyDown:
		.Child().ActivateNext()
	case fyne.KeyUp:
		.Child().ActivatePrevious()
	case fyne.KeyEnter, fyne.KeyReturn, fyne.KeySpace:
		.Child().TriggerLast()
	}
}

type menuBarItemRenderer struct {
	widget.BaseRenderer
	i          *menuBarItem

func ( *menuBarItemRenderer) ( fyne.Size) {
	 := .padding()

	.text.TextSize = theme.TextSize()
	.text.Color = theme.Color(theme.ColorNameForeground)
	.text.Resize(.text.MinSize())
	.text.Move(fyne.NewPos(.Width/2, .Height/2))

	.background.Resize()
}

func ( *menuBarItemRenderer) () fyne.Size {
	return .text.MinSize().Add(.padding())
}

func ( *menuBarItemRenderer) () {
	.background.CornerRadius = theme.SelectionRadiusSize()
	if .i.active && .i.Parent.active {
		.background.FillColor = theme.Color(theme.ColorNameFocus)
		.background.Show()
	} else if .i.hovered && !.i.Parent.active {
		.background.FillColor = theme.Color(theme.ColorNameHover)
		.background.Show()
	} else {
		.background.Hide()
	}
	.background.Refresh()
	canvas.Refresh(.i)
}

func ( *menuBarItemRenderer) () fyne.Size {
	return fyne.NewSize(theme.InnerPadding()*2, theme.InnerPadding())
}
