type aatFeatureRange struct {
	info       aatFeatureInfo
	start, end int
}

type aatFeatureEvent struct {
	index   int
	start   bool
	feature aatFeatureInfo
}

func ( aatFeatureEvent) ( aatFeatureEvent) bool {
	if .index < .index {
		return true
	} else if .index > .index {
		return false
	} else {
		if !.start && .start {
			return true
		} else if .start && !.start {
			return false
		}
		return .feature.isLess(.feature)
	}
}

type aatFeatureInfo struct {
	type_       aatLayoutFeatureType
	setting     aatLayoutFeatureSelector
	isExclusive bool
}

func ( aatFeatureInfo) () uint32 {
	return uint32(.type_)<<16 | uint32(.setting)
}

const selMask = ^aatLayoutFeatureSelector(1)

func ( aatFeatureInfo) ( aatFeatureInfo) bool {
	if .type_ != .type_ {
		return .type_ < .type_
	}
	if !.isExclusive && (.setting&selMask) != (.setting&selMask) {
		return .setting < .setting
	}
	return false
}
