func (unicodeFuncs) ( rune) uint8 {
	// Reorder SAKOT to ensure it comes after any tone marks.
	if  == 0x1A60 {
		return 254
	}

	// Reorder PADMA to ensure it comes after any vowel marks.
	if  == 0x0FC6 {
		return 254
	}
	/* Reorder TSA -PHRU to reorder before U+0F74 */
	if  == 0x0F39 {
		return 127
	}
	return modifiedCombiningClass[ucd.LookupCombiningClass()]
}

// IsDefaultIgnorable returns `true` for

	// is with regular spacing glyphs, and that's the way fonts are made to work.
	// As such, we make exceptions for those four.
	// Also ignoring U+1BCA0..1BCA3. https://github.com/harfbuzz/harfbuzz/issues/503

		case 0x17:

		case 0x20:
			return 0x200B <=  &&  <= 0x200F ||

		case 0xFE:

// Implement enough of Unicode Graphemes here that shaping
// in reverse-direction wouldn't break graphemes.  Namely,
// we mark all marks and ZWJ and ZWJ,Extended_Pictographic
// sequences as continuations.  The foreach_grapheme()
// macro uses this bit.
//
// https://www.unicode.org/reports/tr29/#Regex_Definitions
func ( *Buffer) () {
	 := .Info
	for  := 0;  < len(); ++ {
		 := [].codepoint
		[].setUnicodeProps()

		/* Marks are already set as continuation by the above line.
		 * Handle Emoji_Modifier and ZWJ-continuation. */
		if [].unicode.generalCategory() == modifierSymbol && (0x1F3FB <=  &&  <= 0x1F3FF) {
			[].setContinuation()
		} else if  != 0 && isRegionalIndicator() {
			/* Regional_Indicators are hairy as hell...
			* https://github.com/harfbuzz/harfbuzz/issues/2265 */
			if isRegionalIndicator([-1].codepoint) && ![-1].isContinuation() {
				[].setContinuation()
			}
		} else if [].isZwj() {
			[].setContinuation()
			if +1 < len(.Info) && uni.isExtendedPictographic([+1].codepoint) {
				++
				[].setUnicodeProps()
				[].setContinuation()
			}
		} else if (0xFF9E <=  &&  <= 0xFF9F) || (0xE0020 <=  &&  <= 0xE007F) {
			// Or part of the Other_Grapheme_Extend that is not marks.
			// As of Unicode 15 that is just:
			//
			// 200C          ; Other_Grapheme_Extend # Cf       ZERO WIDTH NON-JOINER
			// FF9E..FF9F    ; Other_Grapheme_Extend # Lm   [2] HALFWIDTH KATAKANA VOICED SOUND MARK..HALFWIDTH KATAKANA SEMI-VOICED SOUND MARK
			// E0020..E007F  ; Other_Grapheme_Extend # Cf  [96] TAG SPACE..CANCEL TAG
			//
			// ZWNJ is special, we don't want to merge it as there's no need, and keeping
			// it separate results in more granular clusters.
			// Tags are used for Emoji sub-region flag sequences:
			// https://github.com/harfbuzz/harfbuzz/issues/1556
			// Katakana ones were requested:
			// https://github.com/harfbuzz/harfbuzz/issues/3844
			[].setContinuation()
		}
	}
}

	 := uni.generalCategory()
	 := unicodeProp()

	if  >= 0x80 {
		 |= bsfHasNonASCII

		if uni.isDefaultIgnorable() {
			 |= bsfHasDefaultIgnorables
			 |= upropsMaskIgnorable

			} else if  == 0x200D {
				 |= upropsMaskCfZwj
			} else if (0x180B <=  &&  <= 0x180D) ||  == 0x180F {

				 * because although we need to hide them like default-ignorables,
				 * they need to non-ignorable during shaping.  This is similar to
				 * what we do for joiners in Indic-like shapers, but since the
				 * FVSes are GC=Mn, we have use a separate bit to remember them.
				 * Fixes:
				 * https://github.com/harfbuzz/harfbuzz/issues/234 */
				 |= upropsMaskHidden
			} else if 0xE0020 <=  &&  <= 0xE007F {
				/* TAG characters need similar treatment. Fixes:
				 * https://github.com/harfbuzz/harfbuzz/issues/463 */

			} else if  == 0x034F {
				/* COMBINING GRAPHEME JOINER should not be skipped; at least some times.
				 * https://github.com/harfbuzz/harfbuzz/issues/554 */
				 |= bsfHasCGJ
				 |= upropsMaskHidden
			}
		}

		if .isMark() {
			 |= upropsMaskContinuation
			 |= unicodeProp(uni.modifiedCombiningClass()) << 8
		}
	}
