	.addGSUBPause(clearSubstitutionFlags)

	.addGSUBPause(recordPrefUse)

	/* "Orthographic unit shaping group" */
	for ,  := range useBasicFeatures {
		.enableFeatureExt(, ffManualZWJ|ffPerSyllable, 1)
	}

	.addGSUBPause(reorderUse)
	.addGSUBPause(nil)

	/* "Topographical features" */
	for ,  := range useTopographicalFeatures {
		.addFeature()
	}
	.addGSUBPause(nil)

	/* "Standard typographic presentation" */
	for ,  := range useOtherFeatures {
		.enableFeatureExt(, ffManualZWJ, 1)
	}
}
