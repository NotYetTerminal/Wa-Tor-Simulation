func ( *otShapePlanner) ( *otShapePlan,  otShapePlanKey) {
	.props = .props
	.shaper = .shaper
	.map_.compile(&.map_, )

	.fracMask = .map_.getMask1(ot.NewTag('f', 'r', 'a', 'c'))
	.numrMask = .map_.getMask1(ot.NewTag('n', 'u', 'm', 'r'))
	.dnomMask = .map_.getMask1(ot.NewTag('d', 'n', 'o', 'm'))
	.hasFrac = .fracMask != 0 || (.numrMask != 0 && .dnomMask != 0)

	.rtlmMask = .map_.getMask1(ot.NewTag('r', 't', 'l', 'm'))
	.hasVert = .map_.getMask1(ot.NewTag('v', 'e', 'r', 't')) != 0

	 := ot.NewTag('v', 'k', 'r', 'n')
	if .props.Direction.isHorizontal() {
		 = ot.NewTag('k', 'e', 'r', 'n')
	}

	.kernMask, _ = .map_.getMask()
	.requestedKerning = .kernMask != 0
	.trakMask, _ = .map_.getMask(ot.NewTag('t', 'r', 'a', 'k'))
	.requestedTracking = .trakMask != 0

	 := .map_.getFeatureIndex(1, ) != NoFeatureIndex
	 := .shaper.gposTag() != 0 && .shaper.gposTag() != .map_.chosenScript[1]

	// Decide who provides glyph classes. GDEF or Unicode.
	if .tables.GDEF.GlyphClassDef == nil {
		.fallbackGlyphClasses = true
	}

	// Decide who does substitutions. GSUB, morx, or fallback.
	.applyMorx = .applyMorx

	//  Decide who does positioning. GPOS, kerx, kern, or fallback.
	 := .tables.Kerx != nil
	 := !.applyMorx && .tables.GSUB.Lookups != nil
	 := ! && .tables.GPOS.Lookups != nil

	if  && !( && ) {
		.applyKerx = true
	} else if  {
		.applyGpos = true
	}

	if !.applyKerx && (! || !.applyGpos) {
		// apparently Apple applies kerx if GPOS kern was not applied.
		if  {
			.applyKerx = true
		} else if .tables.Kern != nil {
			.applyKern = true
		}
	}

	.applyFallbackKern = !(.applyGpos || .applyKerx || .applyKern)

	.zeroMarks = .scriptZeroMarks && !.applyKerx &&
		(!.applyKern || !hasMachineKerning(.tables.Kern))
	.hasGposMark = .map_.getMask1(ot.NewTag('m', 'a', 'r', 'k')) != 0

	.adjustMarkPositioningWhenZeroing = !.applyGpos && !.applyKerx &&
		(!.applyKern || !hasCrossKerning(.tables.Kern))

	.fallbackMarkPositioning = .adjustMarkPositioningWhenZeroing && .scriptFallbackMarkPositioning

	// If we're using morx shaping, we cancel mark position adjustment because
	// Apple Color Emoji assumes this will NOT be done when forming emoji sequences;
	// https://github.com/harfbuzz/harfbuzz/issues/2967.
	if .applyMorx {
		.adjustMarkPositioningWhenZeroing = false
	}

	// currently we always apply trak.
	.applyTrak = .requestedTracking && !.tables.Trak.IsEmpty()
}

func ( *otShapePlanner) ( []Feature) {
	 := &.map_

	.enableFeature(ot.NewTag('r', 'v', 'r', 'n'))
	.addGSUBPause(nil)

	switch .props.Direction {
	case LeftToRight:
		.enableFeature(ot.NewTag('l', 't', 'r', 'a'))
		.enableFeature(ot.NewTag('l', 't', 'r', 'm'))
	case RightToLeft:
		.enableFeature(ot.NewTag('r', 't', 'l', 'a'))
		.addFeature(ot.NewTag('r', 't', 'l', 'm'))
	}

	/* Automatic fractions. */
	.addFeature(ot.NewTag('f', 'r', 'a', 'c'))
	.addFeature(ot.NewTag('n', 'u', 'm', 'r'))
	.addFeature(ot.NewTag('d', 'n', 'o', 'm'))

	/* Random! */
	.enableFeatureExt(ot.NewTag('r', 'a', 'n', 'd'), ffRandom, otMapMaxValue)

	/* Tracking.  We enable dummy feature here just to allow disabling
	* AAT 'trak' table using features.
	* https://github.com/harfbuzz/harfbuzz/issues/1303 */
	.enableFeatureExt(ot.NewTag('t', 'r', 'a', 'k'), ffHasFallback, 1)

	.enableFeature(ot.NewTag('H', 'a', 'r', 'f')) /* Considered required. */
	.enableFeature(ot.NewTag('H', 'A', 'R', 'F')) /* Considered discretionary. */

	.shaper.collectFeatures()

	.enableFeature(ot.NewTag('B', 'u', 'z', 'z')) /* Considered required. */
	.enableFeature(ot.NewTag('B', 'U', 'Z', 'Z')) /* Considered discretionary. */

	for ,  := range commonFeatures {
		.addFeatureExt(.tag, .flags, 1)
	}

	if .props.Direction.isHorizontal() {
		for ,  := range horizontalFeatures {
			.addFeatureExt(.tag, .flags, 1)
		}
	} else {
		/* We really want to find a 'vert' feature if there's any in the font, no
		 * matter which script/langsys it is listed (or not) under.
		 * See various bugs referenced from:
		 * https://github.com/harfbuzz/harfbuzz/issues/63 */
		.enableFeatureExt(ot.NewTag('v', 'e', 'r', 't'), ffGlobalSearch, 1)
	}

	for ,  := range  {
		 := ffNone
		if .Start == FeatureGlobalStart && .End == FeatureGlobalEnd {
			 = ffGLOBAL
		}
		.addFeatureExt(.Tag, , .Value)
	}

	.shaper.overrideFeatures()
}
