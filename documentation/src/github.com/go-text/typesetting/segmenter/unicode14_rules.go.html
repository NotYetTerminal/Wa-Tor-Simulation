// Apply the Line Breaking Rules and returns the computed break opportunity
// See https://unicode.org/reports/tr14/#BreakingRules
func ( *cursor) () breakOpportunity {
	// start by attributing the break class for the current rune
	.ruleLB1()

	 := .updateNumSequence()

	// add the line break rules in reverse order to override
	// the lower priority rules.
	 := breakEmpty

	.ruleLB30(&)
	.ruleLB30ab(&)
	.ruleLB29To26(&)
	.ruleLB25(&, )
	.ruleLB24To22(&)
	.ruleLB21To9(&)
	.ruleLB8(&)
	.ruleLB7To4(&)

	return 
}

// breakOpportunity is a convenient enum,
// mapped to the LineBreak and MandatoryBreak properties,
// avoiding too many bit operations
type breakOpportunity uint8

const (
	breakEmpty      breakOpportunity = iota // not specified
	breakProhibited                         // no break
	breakAllowed                            // direct break (can always break here)
	breakMandatory                          // break is mandatory (implies breakAllowed)
)

func ( *cursor) ( *breakOpportunity) {
	// (AL | HL | NU) × [OP-[\p{ea=F}\p{ea=W}\p{ea=H}]]
	if (.prevLine == ucd.BreakAL || .prevLine == ucd.BreakHL || .prevLine == ucd.BreakNU) &&
		.line == ucd.BreakOP && !unicode.Is(ucd.LargeEastAsian, .r) {
		* = breakProhibited
	}
	// [CP-[\p{ea=F}\p{ea=W}\p{ea=H}]] × (AL | HL | NU)
	if .prevLine == ucd.BreakCP && !unicode.Is(ucd.LargeEastAsian, .prev) &&
		(.line == ucd.BreakAL || .line == ucd.BreakHL || .line == ucd.BreakNU) {
		* = breakProhibited
	}
}

func ( *cursor) ( *breakOpportunity) {
	// (RI RI)* RI × RI
	if .isPrevLinebreakRIOdd && .line == ucd.BreakRI { // LB30a
		* = breakProhibited
	}

	// LB30b
	// EB × EM
	if .prevLine == ucd.BreakEB && .line == ucd.BreakEM {
		* = breakProhibited
	}
	// [\p{Extended_Pictographic}&\p{Cn}] × EM
	if unicode.Is(ucd.Extended_Pictographic, .prev) && ucd.LookupType(.prev) == nil &&
		.line == ucd.BreakEM {
		* = breakProhibited
	}
}

func ( *cursor) ( *breakOpportunity) {
	,  := .prevLine, .line
	// LB29 : IS × (AL | HL)
	if  == ucd.BreakIS && ( == ucd.BreakAL ||  == ucd.BreakHL) {
		* = breakProhibited
	}
	// LB28 : (AL | HL) × (AL | HL)
	if ( == ucd.BreakAL ||  == ucd.BreakHL) && ( == ucd.BreakAL ||  == ucd.BreakHL) {
		* = breakProhibited
	}
	// LB27
	// (JL | JV | JT | H2 | H3) × PO
	if ( == ucd.BreakJL ||  == ucd.BreakJV ||  == ucd.BreakJT ||  == ucd.BreakH2 ||  == ucd.BreakH3) &&
		 == ucd.BreakPO {
		* = breakProhibited
	}
	// PR × (JL | JV | JT | H2 | H3)
	if  == ucd.BreakPR &&
		( == ucd.BreakJL ||  == ucd.BreakJV ||  == ucd.BreakJT ||  == ucd.BreakH2 ||  == ucd.BreakH3) {
		* = breakProhibited
	}
	// LB26
	// JL × (JL | JV | H2 | H3)
	if  == ucd.BreakJL &&
		( == ucd.BreakJL ||  == ucd.BreakJV ||  == ucd.BreakH2 ||  == ucd.BreakH3) {
		* = breakProhibited
	}
	// (JV | H2) × (JV | JT)
	if ( == ucd.BreakJV ||  == ucd.BreakH2) && ( == ucd.BreakJV ||  == ucd.BreakJT) {
		* = breakProhibited
	}
	// (JT | H3) × JT
	if ( == ucd.BreakJT ||  == ucd.BreakH3) &&  == ucd.BreakJT {
		* = breakProhibited
	}
}

// we follow other implementations by using the tailoring described
// in Example 7

	,  := .prevLine, .line
	// (PR | PO) × ( OP | HY )? NU
	if ( == ucd.BreakPR ||  == ucd.BreakPO) &&  == ucd.BreakNU {
		* = breakProhibited

func ( *cursor) ( *breakOpportunity) {
	,  := .prevLine, .line
	// LB24
	// (PR | PO) × (AL | HL)
	if ( == ucd.BreakPR ||  == ucd.BreakPO) && ( == ucd.BreakAL ||  == ucd.BreakHL) {
		* = breakProhibited
	}
	// (AL | HL) × (PR | PO)
	if ( == ucd.BreakAL ||  == ucd.BreakHL) && ( == ucd.BreakPR ||  == ucd.BreakPO) {
		* = breakProhibited
	}
	// LB23
	// (AL | HL) × NU
	if ( == ucd.BreakAL ||  == ucd.BreakHL) &&  == ucd.BreakNU {
		* = breakProhibited
	}
	// NU × (AL | HL)
	if  == ucd.BreakNU && ( == ucd.BreakAL ||  == ucd.BreakHL) {
		* = breakProhibited
	}
	// LB23a
	// PR × (ID | EB | EM)
	if  == ucd.BreakPR && ( == ucd.BreakID ||  == ucd.BreakEB ||  == ucd.BreakEM) {
		* = breakProhibited
	}
	// (ID | EB | EM) × PO
	if ( == ucd.BreakID ||  == ucd.BreakEB ||  == ucd.BreakEM) &&  == ucd.BreakPO {
		* = breakProhibited
	}

	// LB22 : × IN
	if  == ucd.BreakIN {
		* = breakProhibited
	}
}

func ( *cursor) ( *breakOpportunity) {
	,  := .prevLine, .line
	// LB21
	// × BA
	// × HY
	// × NS
	// BB ×
	if  == ucd.BreakBA ||  == ucd.BreakHY ||  == ucd.BreakNS ||  == ucd.BreakBB {
		* = breakProhibited
	}
	// LB21a : HL (HY | BA) ×
	if .prevPrevLine == ucd.BreakHL &&
		( == ucd.BreakHY ||  == ucd.BreakBA) {
		* = breakProhibited
	}
	// LB21b : SY × HL
	if  == ucd.BreakSY &&  == ucd.BreakHL {
		* = breakProhibited
	}
	// LB20
	// ÷ CB
	// CB ÷
	if  == ucd.BreakCB ||  == ucd.BreakCB {
		* = breakAllowed
	}
	// LB19
	// × QU
	// QU ×
	if  == ucd.BreakQU ||  == ucd.BreakQU {
		* = breakProhibited
	}
	// LB18 : SP ÷
	if  == ucd.BreakSP {
		* = breakAllowed
	}
	// LB17 : B2 SP* × B2
	if .beforeSpaces == ucd.BreakB2 &&  == ucd.BreakB2 {
		* = breakProhibited
	}
	// LB16 : (CL | CP) SP* × NS
	if (.beforeSpaces == ucd.BreakCL || .beforeSpaces == ucd.BreakCP) &&  == ucd.BreakNS {
		* = breakProhibited
	}
	// LB15 : QU SP* × OP
	if .beforeSpaces == ucd.BreakQU &&  == ucd.BreakOP {
		* = breakProhibited
	}
	// LB14 : OP SP* ×
	if .beforeSpaces == ucd.BreakOP {
		* = breakProhibited
	}

	// rule LB13, with the tailoring described in Example 7
	// × EX
	if  == ucd.BreakEX {
		* = breakProhibited
	}
	// [^NU] × CL
	// [^NU] × CP
	// [^NU] × IS
	// [^NU] × SY
	if  != ucd.BreakNU &&
		( == ucd.BreakCL ||  == ucd.BreakCP ||  == ucd.BreakIS ||  == ucd.BreakSY) {
		* = breakProhibited
	}
	// LB12 : GL ×
	if  == ucd.BreakGL {
		* = breakProhibited
	}
	// LB12a : [^SP BA HY] × GL
	if ( != ucd.BreakSP &&  != ucd.BreakBA &&  != ucd.BreakHY) &&
		 == ucd.BreakGL {
		* = breakProhibited
	}
	// LB11
	// × WJ
	// WJ ×
	if  == ucd.BreakWJ ||  == ucd.BreakWJ {
		* = breakProhibited
	}

	// rule LB9 : "Do not break a combining character sequence"
	// where X is any line break class except BK, CR, LF, NL, SP, or ZW.
	// see also [endIteration]
	if  == ucd.BreakCM ||  == ucd.BreakZWJ {
		if !( == ucd.BreakBK ||  == ucd.BreakCR ||  == ucd.BreakLF ||
			 == ucd.BreakNL ||  == ucd.BreakSP ||  == ucd.BreakZW) {
			* = breakProhibited
		}
	}
}

func ( *cursor) ( *breakOpportunity) {
	// rule LB8 : ZW SP* ÷
	if .beforeSpaces == ucd.BreakZW {
		* = breakAllowed
	}
	// rule LB8a : ZWJ ×
	// there is a catch here : prevLine is not always exactly
	// the class at index i-1, because of rules LB9 and LB10
	// however, rule LB8a applies before LB9 and LB10, meaning
	// we need to use the real class
	if unicode.Is(ucd.BreakZWJ, .prev) {
		* = breakProhibited
	}
}

func ( *cursor) ( *breakOpportunity) {
	// LB7
	// × SP
	// × ZW
	if .line == ucd.BreakSP || .line == ucd.BreakZW {
		* = breakProhibited
	}
	// LB6 : × ( BK | CR | LF | NL )
	if .line == ucd.BreakBK || .line == ucd.BreakCR || .line == ucd.BreakLF || .line == ucd.BreakNL {
		* = breakProhibited
	}

	// LB4 and LB5
	// BK !
	// CR !
	// LF !
	// NL !
	// (CR × LF is actually handled in rule LB6)
	if .prevLine == ucd.BreakBK || (.prevLine == ucd.BreakCR && .r != '\n') ||
		.prevLine == ucd.BreakLF || .prevLine == ucd.BreakNL {
		* = breakMandatory
	}
}

// apply rule LB1 to resolve break classses AI, SG, XX, SA and CJ.
// We use the default values specified in https://unicode.org/reports/tr14/#BreakingRules.
func ( *cursor) () {
	switch .line {
	case ucd.BreakAI, ucd.BreakSG, ucd.BreakXX:
		.line = ucd.BreakAL
	case ucd.BreakSA:
		 := ucd.LookupType(.r)
		if  == unicode.Mn ||  == unicode.Mc {
			.line = ucd.BreakCM
		} else {
			.line = ucd.BreakAL
		}
	case ucd.BreakCJ:
		.line = ucd.BreakNS
	}
}

type numSequenceState uint8

const (
	noNumSequence numSequenceState = iota // we are not in a sequence
	inNumSequence                         // we are in NU (NU | SY | IS)*
	seenCloseNum                          // we are at NU (NU | SY | IS)* (CL | CP)?
)

// update the `numSequence` state used for rule LB25
// and returns true if we matched one
func ( *cursor) () bool {
	// note that rule LB9 also apply : (CM|ZWJ) do not change
	// the flag
	if .line == ucd.BreakCM || .line == ucd.BreakZWJ {
		return false
	}

	switch .numSequence {
	case noNumSequence:
		if .line == ucd.BreakNU { // start a sequence
			.numSequence = inNumSequence
		}
		return false
	case inNumSequence:
		switch .line {
		case ucd.BreakNU, ucd.BreakSY, ucd.BreakIS:
			// NU (NU | SY | IS)* × (NU | SY | IS) : the sequence continue
			return true
		case ucd.BreakCL, ucd.BreakCP:
			// NU (NU | SY | IS)* × (CL | CP)
			.numSequence = seenCloseNum
			return true
		case ucd.BreakPO, ucd.BreakPR:
			// NU (NU | SY | IS)* × (PO | PR) : close the sequence
			.numSequence = noNumSequence
			return true
		default:
			.numSequence = noNumSequence
			return false
		}
	case seenCloseNum:
		.numSequence = noNumSequence // close the sequence anyway
		if .line == ucd.BreakPO || .line == ucd.BreakPR {
			// NU (NU | SY | IS)* (CL | CP) × (PO | PR)
			return true
		}
		return false
	default:
		panic("exhaustive switch")
	}
}

// startIteration updates the cursor properties, setting the current
// rune to text[i].
// Some properties depending on the context are rather
// updated in the previous `endIteration` call.
func ( *cursor) ( []rune,  int) {
	.prev = .r
	if  < len() {
		.r = []
	} else {
		.r = paragraphSeparator
	}
	if  == len() {
		.next = 0
	} else if  == len()-1 {
		// we fill in the last element of `attrs` by assuming
		// there's a paragraph separators off the end of text
		.next = paragraphSeparator
	} else {
		.next = [+1]
	}

	// query general unicode properties for the current rune

	.isExtentedPic = unicode.Is(ucd.Extended_Pictographic, .r)

	.prevGrapheme = .grapheme
	.grapheme = ucd.LookupGraphemeBreakClass(.r)

	if .word != ucd.WordBreakExtendFormat {
		.prevPrevWord = .prevWord
		.prevWord = .word
		.prevWordNoExtend =  - 1
	}
	.word = ucd.LookupWordBreakClass(.r)

	// prevPrevLine and prevLine are handled in endIteration
	.line = .nextLine // avoid calling LookupLineBreakClass twice
	.nextLine = ucd.LookupLineBreakClass(.next)
}

// end the current iteration, computing some of the properties
// required for the next rune and respecting rule LB9 and LB10
func ( *cursor) ( bool) {
	// start by handling rule LB9 and LB10
	if .line == ucd.BreakCM || .line == ucd.BreakZWJ {
		 := .prevLine == ucd.BreakBK ||
			.prevLine == ucd.BreakCR ||
			.prevLine == ucd.BreakLF ||
			.prevLine == ucd.BreakNL ||
			.prevLine == ucd.BreakSP ||
			.prevLine == ucd.BreakZW
		if  ||  { // Rule LB10
			.prevLine = ucd.BreakAL
		} // else rule LB9 : ignore the rune for prevLine and prevPrevLine

	} else { // regular update
		.prevPrevLine = .prevLine
		.prevLine = .line
	}

	// keep track of the rune before the spaces
	if .prevLine != ucd.BreakSP {
		.beforeSpaces = .prevLine
	}

	// update RegionalIndicator parity used for LB30a
	if .line == ucd.BreakRI {
		.isPrevLinebreakRIOdd = !.isPrevLinebreakRIOdd
	} else if !(.line == ucd.BreakCM || .line == ucd.BreakZWJ) { // beware of the rule LB9: (CM|ZWJ) ignore the update
		.isPrevLinebreakRIOdd = false
	}
}
