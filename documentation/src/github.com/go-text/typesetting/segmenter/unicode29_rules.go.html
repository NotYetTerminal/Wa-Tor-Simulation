// Apply the Grapheme_Cluster_Boundary_Rules and returns a true if we are
// at a grapheme break.
// See https://unicode.org/reports/tr29/#Grapheme_Cluster_Boundary_Rules
func ( *cursor) () bool {
	 := .updatePictoSequence()    // apply rule GB11
	 := .updateGraphemeRIOdd() // apply rule GB12 and GB13

	,  := .prevGrapheme, .grapheme
	if .r == '\n' && .prev == '\r' {
		return false // Rule GB3
	} else if  == ucd.GraphemeBreakControl ||  == ucd.GraphemeBreakCR ||  == ucd.GraphemeBreakLF ||
		 == ucd.GraphemeBreakControl ||  == ucd.GraphemeBreakCR ||  == ucd.GraphemeBreakLF {
		return true // Rules GB4 && GB5
	} else if  == ucd.GraphemeBreakL &&
		( == ucd.GraphemeBreakL ||  == ucd.GraphemeBreakV ||  == ucd.GraphemeBreakLV ||  == ucd.GraphemeBreakLVT) { // rule GB6
		return false
	} else if ( == ucd.GraphemeBreakLV ||  == ucd.GraphemeBreakV) && ( == ucd.GraphemeBreakV ||  == ucd.GraphemeBreakT) {
		return false // rule GB7
	} else if ( == ucd.GraphemeBreakLVT ||  == ucd.GraphemeBreakT) &&  == ucd.GraphemeBreakT {
		return false // rule GB8
	} else if  == ucd.GraphemeBreakExtend ||  == ucd.GraphemeBreakZWJ {
		return false // Rule GB9
	} else if  == ucd.GraphemeBreakSpacingMark {
		return false // Rule GB9a
	} else if  == ucd.GraphemeBreakPrepend {
		return false // Rule GB9b
	} else if  { // Rule GB11
		return false
	} else if  {
		return false // Rule GB12 && GB13
	}

	return true // Rule GB999
}

// update `isPrevGraphemeRIOdd` used for the rules GB12 and GB13
// and returns `true` if one of them triggered
func ( *cursor) () ( bool) {
	if .grapheme == ucd.GraphemeBreakRegional_Indicator {
		 = .isPrevGraphemeRIOdd
		.isPrevGraphemeRIOdd = !.isPrevGraphemeRIOdd // switch the parity
	} else {
		.isPrevGraphemeRIOdd = false
	}
	return 
}

// see rule GB11
type pictoSequenceState uint8

const (
	noPictoSequence pictoSequenceState = iota // we are not in a sequence
	inPictoExtend                             // we are in (ExtendedPic)(Extend*) pattern
	seenPictoZWJ                              // we have seen (ExtendedPic)(Extend*)(ZWJ)
)

// update the `pictoSequence` state used for rule GB11 pattern :
// (ExtendedPic)(Extend*)(ZWJ)(ExtendedPic)
// and returns true if we matched one
func ( *cursor) () bool {
	switch .pictoSequence {
	case noPictoSequence:
		// we are not in a sequence yet, start it if we have an ExtendedPic
		if .isExtentedPic {
			.pictoSequence = inPictoExtend
		}
		return false
	case inPictoExtend:
		if .grapheme == ucd.GraphemeBreakExtend {
			// continue the sequence with an Extend rune
		} else if .grapheme == ucd.GraphemeBreakZWJ {
			// close the variable part of the sequence with (ZWJ)
			.pictoSequence = seenPictoZWJ
		} else {
			// stop the sequence
			.pictoSequence = noPictoSequence
		}
		return false
	case seenPictoZWJ:
		// trigger GB11 if we have an ExtendedPic,
		// and reset the sequence
		if .isExtentedPic {
			.pictoSequence = inPictoExtend
			return true
		}
		.pictoSequence = noPictoSequence
		return false
	default:
		panic("exhaustive switch")
	}
}

// update `isPrevWordRIOdd` used for the rules WB15 and WB16
// and returns `true` if one of them triggered
func ( *cursor) () ( bool) {
	if .word == ucd.WordBreakExtendFormat {
		return false // skip
	}

	if .word == ucd.WordBreakRegional_Indicator {
		 = .isPrevWordRIOdd
		.isPrevWordRIOdd = !.isPrevWordRIOdd // switch the parity
	} else {
		.isPrevWordRIOdd = false
	}
	return 
}

// Apply the Word_Boundary_Rules and returns true if we are at a
// word boundary.
// removePrevNoExtend is true if the index [prevWordNoExtend]
// should be marked as NOT being a word boundary
// See https://unicode.org/reports/tr29/#Word_Boundary_Rules
func ( *cursor) ( int) (,  bool) {
	 := .updateWordRIOdd()

	, ,  := .prevPrevWord, .prevWord, .word

	// we apply Rules WB1 and WB2 at the end of the main loop

	 := .prevWordNoExtend == -1

	if .prev == '\u000D' && .r == '\u000A' { // Rule WB3
		 = false
	} else if  == ucd.WordBreakNewlineCRLF &&  {
		// The extra check for prevWordNoExtend is to correctly handle sequences like
		// Newline ÷ Extend × Extend
		// since we have not skipped ExtendFormat yet.
		 = true // Rule WB3a
	} else if  == ucd.WordBreakNewlineCRLF {
		 = true // Rule WB3b
	} else if .prev == 0x200D && .isExtentedPic {
		 = false // Rule WB3c
	} else if  == ucd.WordBreakWSegSpace &&
		 == ucd.WordBreakWSegSpace &&  {
		 = false // Rule WB3d
	} else if  == ucd.WordBreakExtendFormat {
		 = false // Rules WB4
	} else if ( == ucd.WordBreakALetter ||  == ucd.WordBreakHebrew_Letter ||  == ucd.WordBreakNumeric) &&
		( == ucd.WordBreakALetter ||  == ucd.WordBreakHebrew_Letter ||  == ucd.WordBreakNumeric) {
		 = false // Rules WB5, WB8, WB9, WB10
	} else if  == ucd.WordBreakKatakana &&  == ucd.WordBreakKatakana {
		 = false // Rule WB13
	} else if ( == ucd.WordBreakALetter ||
		 == ucd.WordBreakHebrew_Letter ||
		 == ucd.WordBreakNumeric ||
		 == ucd.WordBreakKatakana ||
		 == ucd.WordBreakExtendNumLet) &&
		 == ucd.WordBreakExtendNumLet {
		 = false // Rule WB13a
	} else if  == ucd.WordBreakExtendNumLet &&
		( == ucd.WordBreakALetter ||  == ucd.WordBreakHebrew_Letter ||  == ucd.WordBreakNumeric ||
			 == ucd.WordBreakKatakana) {
		 = false // Rule WB13b
	} else if ( == ucd.WordBreakALetter ||  == ucd.WordBreakHebrew_Letter) &&
		( == ucd.WordBreakMidLetter ||  == ucd.WordBreakMidNumLet ||  == ucd.WordBreakSingle_Quote) &&
		( == ucd.WordBreakALetter ||  == ucd.WordBreakHebrew_Letter) {
		 = true // Rule WB6
		 = false    // Rule WB7
	} else if  == ucd.WordBreakHebrew_Letter &&  == ucd.WordBreakSingle_Quote {
		 = false // Rule WB7a
	} else if  == ucd.WordBreakHebrew_Letter && .prev == 0x0022 &&
		 == ucd.WordBreakHebrew_Letter {
		 = true // Rule WB7b
		 = false    // Rule WB7c
	} else if ( == ucd.WordBreakNumeric &&  == ucd.WordBreakNumeric) &&
		( == ucd.WordBreakMidNum ||  == ucd.WordBreakMidNumLet ||
			 == ucd.WordBreakSingle_Quote) {
		 = false    // Rule WB11
		 = true // Rule WB12
	} else if  {
		 = false // Rule WB15 and WB16
	} else {
		 = true // Rule WB999
	}

	return , 
}
