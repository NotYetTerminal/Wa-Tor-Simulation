// rejects several extensions which are for sure not supported font files
// return `true` is the file should be ignored
func ( string) bool {
	// ignore hidden file
	if  == "" || [0] == '.' {
		return true
	} else if strings.HasSuffix(, ".enc.gz") || // encodings
		strings.HasSuffix(, ".afm") || // metrics (ascii)
		strings.HasSuffix(, ".pfm") || // metrics (binary)
		strings.HasSuffix(, ".dir") || // summary
		strings.HasSuffix(, ".scale") ||
		strings.HasSuffix(, ".alias") ||
		strings.HasSuffix(, ".pcf") || strings.HasSuffix(, ".pcf.gz") || // Bitmap
		strings.HasSuffix(, ".pfb") /* Type1 */ {
		return true
	}

	return false
}
