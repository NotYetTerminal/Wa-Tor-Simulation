// LineMetric returns the metric identified by `metric` (in fonts units).
func ( *Face) ( LineMetric) float32 {
	switch  {
	case UnderlinePosition:
		return .post.underlinePosition + .mvar.getVar(tagUnderlineOffset, .coords)
	case UnderlineThickness:
		return .post.underlineThickness + .mvar.getVar(tagUnderlineSize, .coords)
	case StrikethroughPosition:
		return float32(.os2.yStrikeoutPosition) + .mvar.getVar(tagStrikeoutOffset, .coords)
	case StrikethroughThickness:
		return float32(.os2.yStrikeoutSize) + .mvar.getVar(tagStrikeoutSize, .coords)
	case SuperscriptEmYSize:
		return float32(.os2.ySuperscriptYSize) + .mvar.getVar(tagSuperscriptYSize, .coords)
	case SuperscriptEmXOffset:
		return float32(.os2.ySuperscriptXOffset) + .mvar.getVar(tagSuperscriptXOffset, .coords)
	case SubscriptEmYSize:
		return float32(.os2.ySubscriptYSize) + .mvar.getVar(tagSubscriptYSize, .coords)
	case SubscriptEmYOffset:
		return float32(.os2.ySubscriptYOffset) + .mvar.getVar(tagSubscriptYOffset, .coords)
	case SubscriptEmXOffset:
		return float32(.os2.ySubscriptXOffset) + .mvar.getVar(tagSubscriptXOffset, .coords)
	case CapHeight:
		return float32(.os2.sCapHeight) + .mvar.getVar(tagCapHeight, .coords)
	case XHeight:
		return float32(.os2.sxHeigh) + .mvar.getVar(tagXHeight, .coords)
	default:
		return 0
	}
}
