// type2CharstringHandler implements operators needed to fetch Type2 charstring metrics
type type2CharstringHandler struct {
	cs ps.CharstringReader

	// found in private DICT, needed since we can't differenciate
	// no width set from 0 width
	// `width` must be initialized to default width
	nominalWidthX float64
	width         float64
}

func (type2CharstringHandler) () ps.Context { return ps.Type2Charstring }

func ( *type2CharstringHandler) ( *ps.Machine,  ps.Operator) error {
	var  error
	if !.IsEscaped {
		switch .Operator {
		case 11: // return
			return .Return() // do not clear the arg stack
		case 14: // endchar
			if .ArgStack.Top > 0 { // width is optional
				.width = .nominalWidthX + .ArgStack.Vals[0]
			}
			.cs.ClosePath()
			return ps.ErrInterrupt
		case 10: // callsubr
			return ps.LocalSubr() // do not clear the arg stack
		case 29: // callgsubr
			return ps.GlobalSubr() // do not clear the arg stack
		case 21: // rmoveto
			if .ArgStack.Top > 2 { // width is optional
				.width = .nominalWidthX + .ArgStack.Vals[0]
			}
			 = .cs.Rmoveto()
		case 22: // hmoveto
			if .ArgStack.Top > 1 { // width is optional
				.width = .nominalWidthX + .ArgStack.Vals[0]
			}
			 = .cs.Hmoveto()
		case 4: // vmoveto
			if .ArgStack.Top > 1 { // width is optional
				.width = .nominalWidthX + .ArgStack.Vals[0]
			}
			 = .cs.Vmoveto()
		case 1, 18: // hstem, hstemhm
			.cs.Hstem()
		case 3, 23: // vstem, vstemhm
			.cs.Vstem()
		case 19, 20: // hintmask, cntrmask
			// variable number of arguments, but always even
			// for xxxmask, if there are arguments on the stack, then this is an impliied stem
			if .ArgStack.Top&1 != 0 {
				.width = .nominalWidthX + .ArgStack.Vals[0]
			}
			.cs.Hintmask()
			// the stack is managed by the previous call
			return nil

		case 5: // rlineto
			.cs.Rlineto()
		case 6: // hlineto
			.cs.Hlineto()
		case 7: // vlineto
			.cs.Vlineto()
		case 8: // rrcurveto
			.cs.Rrcurveto()
		case 24: // rcurveline
			 = .cs.Rcurveline()
		case 25: // rlinecurve
			 = .cs.Rlinecurve()
		case 26: // vvcurveto
			.cs.Vvcurveto()
		case 27: // hhcurveto
			.cs.Hhcurveto()
		case 30: // vhcurveto
			.cs.Vhcurveto()
		case 31: // hvcurveto
			.cs.Hvcurveto()
		default:
			// no other operands are allowed before the ones handled above

		}
	} else {

		case 34: // hflex
			 = .cs.Hflex()
		case 35: // flex
			 = .cs.Flex()
		case 36: // hflex1
			 = .cs.Hflex1()
		case 37: // flex1
			 = .cs.Flex1()
		default:
			// no other operands are allowed before the ones handled above
			 = fmt.Errorf("invalid operator %s in charstring", )
		}
	}
	.ArgStack.Clear()

}
