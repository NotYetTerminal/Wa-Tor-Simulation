// Read a 40-byte BITMAPINFOHEADER.
// Use of this function does not imply that the entire header is 40 bytes.
// We may just be decoding the first 40 bytes of a 108- or 124-byte header.
func ( *decoder,  []byte,  bool) error {
	.width = int(int32(getDWORD([4:8])))
	.height = int(int32(getDWORD([8:12])))
	if .height < 0 {

		.height = -.height
	}
	.bitCount = int(getWORD([14:16]))
	if  {
		return nil
	}

	if len() >= 20 {
		.biCompression = getDWORD([16:20])
	}

	if .biCompression == bI_BITFIELDS && .headerSize == 40 && .bitCount != 1 {
		.hasBitFieldsSegment = true
		.bitFieldsSegmentSize = 12
	}

	if .biCompression == bI_RGB {
		if .bitCount == 16 {
			// Default bitfields for 16-bit images:
			.recordBitFields(0x7c00, 0x03e0, 0x001f, 0)
		} else if .bitCount == 32 {
			// Default bitfields for 32-bit images:
			.recordBitFields(0x00ff0000, 0x0000ff00, 0x000000ff, 0)
		}
	}

	var  uint32
	if len() >= 36 {
		 = getDWORD([32:36])
	}
	if  > 10000 {
		return FormatError(fmt.Sprintf("bad palette size %d", ))
	}

	// Figure out how many colors (that we care about) are in the palette.
	if .bitCount >= 1 && .bitCount <= 8 {
		if  == 0 ||  > (1<<uint(.bitCount)) {
			.srcPalNumEntries = 1 << uint(.bitCount)
		} else {
			.srcPalNumEntries = int()

	} else {

	}

	.srcPalBytesPerEntry = 4

	if .headerSize == 64 && .srcPalNumEntries > 0 {
		// A hack to allow (invalid?) OS/2v2 BMPs that have 3 bytes per palette
		// entry instead of 4.
		if 14+.headerSize+uint32(.bitFieldsSegmentSize)+3*uint32(.srcPalNumEntries) ==
			.bfOffBits {
			.srcPalBytesPerEntry = 3
		}
	}

	return nil
}

func ( *decoder) ( io.Reader,  bool) (image.Image, error) {
	var  error

	// Read the FILEHEADER and INFOHEADER.
	 = .readHeaders(false)
	if  != nil {
		return nil, 
	}

	// Make sure bitcount and "compression" are valid and compatible.
	switch .biCompression {
	case bI_RGB:
		// We allow bitCount=2 because Windows CE defines it to be valid
		// (at least if headerSize=40).
		if .bitCount != 1 && .bitCount != 2 && .bitCount != 4 && .bitCount != 8 &&
			.bitCount != 16 && .bitCount != 24 && .bitCount != 32 {
			return nil, FormatError(fmt.Sprintf("bad bit count %d", .bitCount))
		}
	case bI_RLE4:
		if .bitCount != 4 {
			return nil, FormatError(fmt.Sprintf("bad RLE4 bit count %d", .bitCount))
		}
	case bI_RLE8:
		if .bitCount != 8 {
			return nil, FormatError(fmt.Sprintf("bad RLE8 bit count %d", .bitCount))
		}
	case bI_BITFIELDS:
		if .bitCount == 1 {
			return nil, UnsupportedError("Huffman 1D compression")
		} else if .bitCount != 16 && .bitCount != 32 {
			return nil, FormatError(fmt.Sprintf("bad BITFIELDS bit count %d", .bitCount))
		}
	default:
		return nil, UnsupportedError(fmt.Sprintf("compression or image type %d", .biCompression))
	}

	// Assuming 'int' is 32 bits, an NRGBA image can't handle more than (2^31-1)/4
	// pixels. This test is more conservative than it could be.
	if .width > 46340 || .height > 46340 || .width*.height >= 0x20000000 {
		return nil, UnsupportedError("dimensions too large")
	}

	// Read the BITFIELDS segment, if present.
	if .hasBitFieldsSegment {
		 = .readBitFieldsSegment()
		if  != nil {
			return nil, 
		}
	}

	// Read the palette, if present.
	if .srcPalNumEntries > 0 {
		 = .readPalette()
		if  != nil {
			return nil, 
		}
	}

	if  {
		return nil, nil
	}

	// Create the target image.
	if .dstHasPalette {
		.img_Paletted = image.NewPaletted(image.Rect(0, 0, .width, .height), .dstPalette)
	} else {
		.img_NRGBA = image.NewNRGBA(image.Rect(0, 0, .width, .height))
	}

	// Skip over any unused space preceding the bitmap bits.
	 = .readGap()
	if  != nil {
		return nil, 
	}

	// Read the bitmap bits.
	if .biCompression == bI_RLE4 || .biCompression == bI_RLE8 {
		 = .readBitsRLE()
	} else {
		 = .readBitsUncompressed()
	}
	if  != nil {
		return nil, 
	}

	if .dstHasPalette {
		return .img_Paletted, nil
	}
	return .img_NRGBA, nil
}
