// Write the BITMAPINFOHEADER structure to a slice[40] or [124].
func ( *encoder) ( []byte) {
	setDWORD([0:4], uint32(.headerSize))
	setDWORD([4:8], uint32(.width))
	setDWORD([8:12], uint32(.height))
	setWORD([12:14], 1) // biPlanes
	setWORD([14:16], uint16(.dstBitCount))
	if .writeAlpha {
		setWORD([16:20], 3) // "Compression" = BI_BITFIELDS
	}
	setDWORD([20:24], uint32(.dstBitsSize))
	if .opts.densitySet {
		setDWORD([24:28], uint32(.opts.xDens))
		setDWORD([28:32], uint32(.opts.yDens))
	} else {
		setDWORD([24:28], 2835)
		setDWORD([28:32], 2835)
	}
	setDWORD([32:36], uint32(.nColors))

	if len() == 124 {
		// Set V5 header fields
		setDWORD([40:44], 0x00ff0000) // RedMask
		setDWORD([44:48], 0x0000ff00) // GreenMask
		setDWORD([48:52], 0x000000ff) // BlueMask
		setDWORD([52:56], 0xff000000) // AlphaMask
		setDWORD([56:60], 0x73524742) // CSType = sRGB
		setDWORD([108:112], 4)        // Intent = IMAGES (perceptual)
	}
}

// If the image can be written as a paletted image, sets e.writePaletted
// to true, and sets related fields.
func ( *encoder) () {
	if .writeAlpha {
		return
	}

	switch .m.(type) {
	case *image.Paletted:
		.m_AsPaletted = .m.(*image.Paletted)
		.nColors = len(.m_AsPaletted.Palette)
		if .nColors < 1 || .nColors > 256 {
			.m_AsPaletted = nil
			.nColors = 0
			return
		}
		.writePaletted = true
	case *image.Gray, *image.Gray16:
		.srcIsGray = true
		.writePaletted = true
		.nColors = 256
	}
}

// Plot out the structure of the file that we're going to write.
func ( *encoder) () error {
	.srcBounds = .m.Bounds()
	.width = .srcBounds.Dx()
	.height = .srcBounds.Dy()

	if .opts.supportTrns && !.srcIsOpaque() {
		.writeAlpha = true
		.headerSize = 124
	} else {
		.headerSize = 40
	}

	.checkPaletted()
	if .writePaletted {
		if .nColors <= 2 {
			.dstBitCount = 1
		} else if .nColors <= 16 {
			.dstBitCount = 4
		} else {
			.dstBitCount = 8
		}
	} else {
		if .writeAlpha {
			.dstBitCount = 32
		} else {
			.dstBitCount = 24
		}
	}
	.dstStride = ((.width*.dstBitCount + 31) / 32) * 4
	.dstBitsOffset = 14 + .headerSize + 4*.nColors
	.dstBitsSize = .height * .dstStride
	.dstFileSize = .dstBitsOffset + .dstBitsSize
	return nil
}
