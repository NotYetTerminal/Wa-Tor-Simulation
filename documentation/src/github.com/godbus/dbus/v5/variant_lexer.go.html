func ( *varLexer) lexState {
	for {
		 := .next()
		switch {
		case  == -1:
			.emit(tokEOF)
			return nil
		case  == '[':
			.emit(tokArrayStart)
		case  == ']':
			.emit(tokArrayEnd)
		case  == '{':
			.emit(tokDictStart)
		case  == '}':
			.emit(tokDictEnd)
		case  == '<':
			.emit(tokVariantStart)
		case  == '>':
			.emit(tokVariantEnd)
		case  == ':':
			.emit(tokColon)
		case  == ',':
			.emit(tokComma)
		case  == '\'' ||  == '"':
			.backup()
			return varLexString
		case  == '@':
			.backup()
			return varLexType
		case unicode.IsSpace():
			.ignore()
		case unicode.IsNumber() ||  == '+' ||  == '-':
			.backup()
			return varLexNumber
		case  == 'b':
			 := .start
			if  := .peek();  == '"' ||  == '\'' {
				return varLexByteString
			}
			// not a byte string; try to parse it as a type or bool below
			.pos =  + 1
			.width = 1
			fallthrough
		default:
			// either a bool or a type. Try bools first.
			.backup()
			if .pos+4 <= len(.input) {
				if .input[.pos:.pos+4] == "true" {
					.pos += 4
					.emit(tokBool)
					continue
				}
			}
			if .pos+5 <= len(.input) {
				if .input[.pos:.pos+5] == "false" {
					.pos += 5
					.emit(tokBool)
					continue
				}
			}
			// must be a type.
			return varLexType
		}
	}
}
