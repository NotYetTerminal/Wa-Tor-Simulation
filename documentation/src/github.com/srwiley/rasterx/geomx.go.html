// AddTo adds the Path p to q.
func ( Path) ( Adder) {
	for  := 0;  < len(); {
		switch PathCommand([]) {
		case PathMoveTo:
			.Stop(false) // Fixes issues #1 by described by Djadala; implicit close if currently in path.
			.Start(fixed.Point26_6{X: [+1], Y: [+2]})
			 += 3
		case PathLineTo:
			.Line(fixed.Point26_6{X: [+1], Y: [+2]})
			 += 3
		case PathQuadTo:
			.QuadBezier(fixed.Point26_6{X: [+1], Y: [+2]}, fixed.Point26_6{X: [+3], Y: [+4]})
			 += 5
		case PathCubicTo:
			.CubeBezier(fixed.Point26_6{X: [+1], Y: [+2]},
				fixed.Point26_6{X: [+3], Y: [+4]}, fixed.Point26_6{X: [+5], Y: [+6]})
			 += 7
		case PathClose:
			.Stop(true)
			++
		default:
			panic("AddTo: bad path")
		}
	}
	.Stop(false)
}
