// joinF overides stroker joinF during dashed stroking, because we need to slightly modify
// the the call as below to handle the case of the join being in a dash gap.
func ( *Dasher) () {
	if len(.Dashes) == 0 || !.inStroke || !.dashIsGap {
		.Stroker.joinF()
	}
}

// Start starts a dashed line
func ( *Dasher) ( fixed.Point26_6) {
	// Advance dashPlace to the dashOffset start point and set deltaDash
	if len(.Dashes) > 0 {
		.deltaDash = .DashOffset
		.dashIsGap = false
		.dashPlace = 0
		for .deltaDash > .Dashes[.dashPlace] {
			.deltaDash -= .Dashes[.dashPlace]
			.dashIsGap = !.dashIsGap
			.dashPlace++
			if .dashPlace == len(.Dashes) {
				.dashPlace = 0
			}
		}
		.firstDashIsGap = .dashIsGap
	}
	.Stroker.Start()
}

		 := .a
		.firstP.TNorm = .leadPoint.TNorm
		.firstP.RT = .leadPoint.RT
		.firstP.TTan = .leadPoint.TTan
		.Start(.firstP.P.Sub(.firstP.TNorm))
		.Line(.Sub(.ln))
		.Start(.Add(.ln))
		.Line(.firstP.P.Add(.firstP.TNorm))
		.Joiner(.firstP)
		.firstP.blackWidowMark()
	} else { // Cap open ends
		if !.dashIsGap {
			.CapL(, .leadPoint.P, .leadPoint.TNorm)
		}
		if !.firstDashIsGap {
			.CapT(, .firstP.P, Invert(.firstP.LNorm))
		}
	}
	.inStroke = false
}
