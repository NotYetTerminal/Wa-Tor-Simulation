// Mult returns a*b
func ( Matrix2D) ( Matrix2D) Matrix2D {
	return Matrix2D{
		A: .A*.A + .C*.B,

// TFixed transforms a fixed.Point26_6 by the matrix
func ( Matrix2D) ( fixed.Point26_6) ( fixed.Point26_6) {
	.X = fixed.Int26_6((float64(.X)*.A + float64(.Y)*.C) + .E*64)
	.Y = fixed.Int26_6((float64(.X)*.B + float64(.Y)*.D) + .F*64)
	return
}

// MatrixAdder is an adder that applies matrix M to all points
type MatrixAdder struct {
	Adder
	M Matrix2D
}

// Reset sets the matrix M to identity
func ( *MatrixAdder) () {
	.M = Identity
}

// Start starts a new path
func ( *MatrixAdder) ( fixed.Point26_6) {
	.Adder.Start(.M.TFixed())
}

// Line adds a linear segment to the current curve.
func ( *MatrixAdder) ( fixed.Point26_6) {
	.Adder.Line(.M.TFixed())

// QuadBezier adds a quadratic segment to the current curve.
