// ReadGradAttr reads an SVG gradient attribute
func ( *IconCursor) ( xml.Attr) ( error) {
	switch .Name.Local {
	case "gradientTransform":
		.grad.Matrix,  = .parseTransform(.Value)
	case "gradientUnits":
		switch strings.TrimSpace(.Value) {
		case "userSpaceOnUse":
			.grad.Units = rasterx.UserSpaceOnUse
		case "objectBoundingBox":
			.grad.Units = rasterx.ObjectBoundingBox
		}
	case "spreadMethod":
		switch strings.TrimSpace(.Value) {
		case "pad":
			.grad.Spread = rasterx.PadSpread
		case "reflect":
			.grad.Spread = rasterx.ReflectSpread
		case "repeat":
			.grad.Spread = rasterx.RepeatSpread
		}
	}
	return
}

func ( *IconCursor) ( rasterx.Matrix2D,  string) (rasterx.Matrix2D, error) {
	 := len(.points)

	case "rotate":

		} else if  == 3 {
			 = .Translate(.points[1], .points[2]).
				Rotate(.points[0]*math.Pi/180).

		}

		} else if  == 2 {

		}

		if  == 1 {

		} else {
			return , errParamMismatch
		}
	case "scale":
		if  == 1 {
			 = .Scale(.points[0], 0)
		} else if  == 2 {
			 = .Scale(.points[0], .points[1])
		} else {
			return , errParamMismatch
		}
	case "matrix":
		if  == 6 {
			 = .Mult(rasterx.Matrix2D{
				A: .points[0],
				B: .points[1],
				C: .points[2],
				D: .points[3],
				E: .points[4],
				F: .points[5]})
		} else {
			return , errParamMismatch
		}
	default:
		return , errParamMismatch
	}
	return , nil
}
