// addSeg decodes an SVG seqment string into equivalent raster path commands saved
// in the cursor's Path
func ( *PathCursor) ( string) error {
	// Parse the string describing the numeric points in SVG format
	if  := .GetPoints([1:]);  != nil {
		return 
	}
	 := len(.points)
	 := [0]
	 := false
	switch  {
	case 'z':
		fallthrough
	case 'Z':
		if len(.points) != 0 {
			return errParamMismatch
		}
		if .inPath {
			.Path.Stop(true)
			.placeX = .pathStartX
			.placeY = .pathStartY
			.inPath = false
		}
	case 'm':
		 = true
		fallthrough
	case 'M':
		if !.hasSetsOrMore(2, ) {
			return errParamMismatch
		}
		.pathStartX, .pathStartY = .points[0], .points[1]
		.inPath = true
		.Path.Start(fixed.Point26_6{X: fixed.Int26_6((.pathStartX) * 64), Y: fixed.Int26_6((.pathStartY) * 64)})
		for  := 2;  < -1;  += 2 {
			.Path.Line(fixed.Point26_6{
				X: fixed.Int26_6((.points[]) * 64),
				Y: fixed.Int26_6((.points[+1]) * 64)})
		}
		.placeX = .points[-2]
		.placeY = .points[-1]
	case 'l':
		 = true
		fallthrough
	case 'L':
		if !.hasSetsOrMore(2, ) {
			return errParamMismatch
		}
		for  := 0;  < -1;  += 2 {
			.Path.Line(fixed.Point26_6{
				X: fixed.Int26_6((.points[]) * 64),
				Y: fixed.Int26_6((.points[+1]) * 64)})
		}
		.placeX = .points[-2]
		.placeY = .points[-1]
	case 'v':
		.valsToAbs(.placeY)
		fallthrough
	case 'V':
		if !.hasSetsOrMore(1, false) {
			return errParamMismatch
		}
		for ,  := range .points {
			.Path.Line(fixed.Point26_6{
				X: fixed.Int26_6((.placeX) * 64),
				Y: fixed.Int26_6(() * 64)})
		}
		.placeY = .points[-1]
	case 'h':
		.valsToAbs(.placeX)
		fallthrough
	case 'H':
		if !.hasSetsOrMore(1, false) {
			return errParamMismatch
		}
		for ,  := range .points {
			.Path.Line(fixed.Point26_6{
				X: fixed.Int26_6(() * 64),
				Y: fixed.Int26_6((.placeY) * 64)})
		}
		.placeX = .points[-1]
	case 'q':
		 = true
		fallthrough
	case 'Q':
		if !.hasSetsOrMore(4, ) {
			return errParamMismatch
		}
		for  := 0;  < -3;  += 4 {
			.Path.QuadBezier(
				fixed.Point26_6{
					X: fixed.Int26_6((.points[]) * 64),
					Y: fixed.Int26_6((.points[+1]) * 64)},
				fixed.Point26_6{
					X: fixed.Int26_6((.points[+2]) * 64),
					Y: fixed.Int26_6((.points[+3]) * 64)})
		}
		.cntlPtX, .cntlPtY = .points[-4], .points[-3]
		.placeX = .points[-2]
		.placeY = .points[-1]
	case 't':
		 = true
		fallthrough
	case 'T':
		if !.hasSetsOrMore(2, ) {
			return errParamMismatch
		}
		for  := 0;  < -1;  += 2 {
			.reflectControlQuad()
			.Path.QuadBezier(
				fixed.Point26_6{
					X: fixed.Int26_6((.cntlPtX) * 64),
					Y: fixed.Int26_6((.cntlPtY) * 64)},
				fixed.Point26_6{
					X: fixed.Int26_6((.points[]) * 64),
					Y: fixed.Int26_6((.points[+1]) * 64)})

			.placeY = .points[+1]
		}

		}

			.Path.CubeBezier(

					X: fixed.Int26_6((.points[]) * 64),

					X: fixed.Int26_6((.points[+2]) * 64),
					Y: fixed.Int26_6((.points[+3]) * 64)},

					X: fixed.Int26_6((.points[+4]) * 64),

		}
		.cntlPtX, .cntlPtY = .points[-4], .points[-3]
		.placeX = .points[-2]
		.placeY = .points[-1]
	case 's':
		 = true
		fallthrough
	case 'S':
		if !.hasSetsOrMore(4, ) {
			return errParamMismatch
		}
		for  := 0;  < -3;  += 4 {
			.reflectControlCube()
			.Path.CubeBezier(fixed.Point26_6{
				X: fixed.Int26_6((.cntlPtX) * 64), Y: fixed.Int26_6((.cntlPtY) * 64)},
				fixed.Point26_6{
					X: fixed.Int26_6((.points[]) * 64), Y: fixed.Int26_6((.points[+1]) * 64)},
				fixed.Point26_6{
					X: fixed.Int26_6((.points[+2]) * 64), Y: fixed.Int26_6((.points[+3]) * 64)})
			.lastKey = 
			.cntlPtX, .cntlPtY = .points[], .points[+1]
			.placeX = .points[+2]
			.placeY = .points[+3]
		}
	case 'a', 'A':
		if !.hasSetsOrMore(7, false) {
			return errParamMismatch
		}
		for  := 0;  < -6;  += 7 {
			if  == 'a' {
				.points[+5] += .placeX
				.points[+6] += .placeY
			}
			.AddArcFromA(.points[:])
		}
	default:

			return errCommandUnknown
		}
		if .ErrorMode == WarnErrorMode {
			log.Println("Ignoring svg command " + string())
		}
	}

	return nil
