	stringSignature = dbus.SignatureOfType(reflect.TypeOf(""))

// FromBool is a fast converter from a boolean to its dbus representation.
func ( bool) dbus.Variant {
	return dbus.MakeVariantWithSignature(, boolSignature)
}
